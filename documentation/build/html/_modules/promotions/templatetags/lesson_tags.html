def clean_file_name(file_name):
    if "/" in file_name:
        file_name = file_name.split("/")[-1]

    if "." in file_name:
        file_name = ".".join(file_name.split(".")[:-1])
    if "_" in file_name:
        file_name = "_".join(file_name.split("_")[:-1])
    return file_name
